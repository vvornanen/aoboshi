## Light theme

<ColorPalette>
  <ColorItem title="primary" subtitle="" colors={lightPalette.primary} />
  <ColorItem title="surface" subtitle="" colors={lightPalette.surface} />
  <ColorItem title="outline" subtitle="" colors={lightPalette.outline} />
  <ColorItem
    title="states"
    subtitle="neutral color"
    colors={Object.fromEntries(
      Object.entries(lightPalette.states).filter(
        ([key]) => !key.includes("Primary"),
      ),
    )}
  />
  <ColorItem
    title="states"
    subtitle="primary color"
    colors={Object.fromEntries(
      Object.entries(lightPalette.states).filter(([key]) =>
        key.includes("Primary"),
      ),
    )}
  />
  <ColorItem title="additional" subtitle="" colors={lightPalette.additional} />
</ColorPalette>
